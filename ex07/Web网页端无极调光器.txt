#include <WiFi.h>
#include <WebServer.h>

// ==================== 配置区 ====================
const char* ssid     = "hid39895789的Pura 70";
const char* password = "yys112233";
const int ledPin = 2;  // 板载LED引脚（或接外部LED）

WebServer server(80);
int brightness = 0;     // 当前亮度（0-255）

// ==================== HTML页面 ====================
const char* htmlPage = R"HTML(
<!DOCTYPE html>
<html lang="zh-CN">
<head>
    <meta charset="UTF-8">
    <meta name="viewport" content="width=device-width, initial-scale=1.0">
    <title>ESP32 无极调光器</title>
    <style>
        body {
            font-family: Arial, sans-serif;
            text-align: center;
            margin-top: 100px;
            background-color: #f0f0f0;
        }
        .container {
            background: white;
            padding: 30px;
            border-radius: 10px;
            box-shadow: 0 0 10px rgba(0,0,0,0.1);
            display: inline-block;
        }
        input[type="range"] {
            width: 300px;
            height: 20px;
            margin: 20px 0;
        }
        #value {
            font-size: 24px;
            font-weight: bold;
            color: #333;
        }
    </style>
</head>
<body>
    <div class="container">
        <h1>LED 无极调光器</h1>
        <p>拖动滑块调整亮度：</p>
        <input type="range" id="brightnessSlider" min="0" max="255" value="0">
        <p>当前亮度：<span id="value">0</span></p>
    </div>

    <script>
        const slider = document.getElementById('brightnessSlider');
        const valueDisplay = document.getElementById('value');

        // 监听滑块变化，实时发送请求到ESP32
        slider.addEventListener('input', function() {
            const brightness = this.value;
            valueDisplay.textContent = brightness;
            
            // 发送GET请求到ESP32
            fetch(`/set?brightness=${brightness}`)
                .then(response => response.text())
                .catch(err => console.error('通信错误:', err));
        });
    </script>
</body>
</html>
)HTML";

// ==================== 路由处理函数 ====================
// 根路由：返回调光页面
void handleRoot() {
  server.send(200, "text/html", htmlPage);
}

// 亮度设置路由：解析URL参数并更新LED
void handleSetBrightness() {
  if (server.hasArg("brightness")) {
    brightness = server.arg("brightness").toInt();
    // 限制范围 0-255
    brightness = constrain(brightness, 0, 255);
    analogWrite(ledPin, brightness);
    Serial.print("设置亮度: ");
    Serial.println(brightness);
  }
  server.send(200, "text/plain", "OK");
}

// ==================== Setup初始化 ====================
void setup() {
  Serial.begin(115200);
  pinMode(ledPin, OUTPUT);
  analogWrite(ledPin, brightness);

  // 连接WiFi
  WiFi.begin(ssid, password);
  while (WiFi.status() != WL_CONNECTED) {
    delay(500);
    Serial.print(".");
  }
  Serial.println("");
  Serial.println("WiFi连接成功！");
  Serial.print("IP地址: ");
  Serial.println(WiFi.localIP());

  // 注册路由
  server.on("/", handleRoot);
  server.on("/set", handleSetBrightness);

  // 启动Web服务器
  server.begin();
  Serial.println("Web服务器已启动");
}

// ==================== 主循环 ====================
void loop() {
  server.handleClient();
}
